% --- SLIDE 2: Brief About the Project ---
\begin{frame}{Brief About the Project}
    \textbf{Problem Statement }
    \begin{itemize}
        \item WhatsApp groups create communication chaos with scattered announcements, lost assignments, and unorganized information
    \end{itemize}
    
    \vspace{0.3cm}
    \textbf{Our Solution: ClassHub }
    \textit{A centralized web-based platform featuring: }
    \begin{itemize}
        \item Organized announcements and notifications 
        \item Assignment management with deadline tracking 
        \item Dynamic timetable with real-time updates 
        \item Automated attendance monitoring 
        \item Subject-wise performance analytics 
    \end{itemize}
\end{frame}
